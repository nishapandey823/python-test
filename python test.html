#function for calculating volume shocks
def flag_df(df):

    if ((((df.unshifted-df.shifted)/df.shifted)*100)>10) :
        return True;      
    elif ((((df.unshifted-df.shifted)/df.shifted)*100)<-10):
        return False
    else :
        return np.nan
